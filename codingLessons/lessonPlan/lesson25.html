So far we have learned how to create a canvas, change the colour, change the name and draw shapes on it. Our next task involves making the game more interactive. Now that we know how to draw shapes, we need to make them clickable.

How do we make shapes clickable?

That's the topic of today's lesson. Lesson 24 Solution

import tkinter as tk


class Shape:
    def __init__(self, color):
        self.color = color

    def draw(self, canvas):
        pass


class Square(Shape):
    def __init__(self, color, x, y, side_length):
        super().__init__(color)
        self.x = x
        self.y = y
        self.side_length = side_length

    def draw(self, canvas):
        canvas.create_rectangle(self.x, self.y,
                                self.x + self.side_length,
                                self.y + self.side_length,
                                fill=self.color)


class Triangle(Shape):
    def __init__(self, color, x1, y1, x2, y2, x3, y3):
        super().__init__(color)
        self.x1 = x1
        self.y1 = y1
        self.x2 = x2
        self.y2 = y2
        self.x3 = x3
        self.y3 = y3

    def draw(self, canvas):
        canvas.create_polygon(self.x1, self.y1,
                              self.x2, self.y2,
                              self.x3, self.y3,
                              fill=self.color)


window = tk.Tk()
window.title("Shapes")
canvas = tk.Canvas(window, width=400, height=400)
canvas.pack()

square1 = Square("#FFD1DC", 50, 50, 100)
square1.draw(canvas)
square2 = Square("#FFD1DC", 200, 200, 80)
square2.draw(canvas)

triangle1 = Triangle("#89997F", 150, 100, 100, 200, 200, 200)
triangle1.draw(canvas)
triangle2 = Triangle("#89997F", 250, 250, 300, 350, 350, 250)
triangle2.draw(canvas)
triangle3 = Triangle("#89997F", 100, 300, 50, 350, 150, 350)
triangle3.draw(canvas)

window.mainloop()
